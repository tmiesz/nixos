** Toggle Markdown View
#+begin_src emacs-lisp

(defun w/toggle-markdown-view-mode ()
  "Toggle between 'markdown-view' and 'markdown-view-mode'."
  (interactive)
  (if (eq major-mode 'markdown-view-mode)
      (markdown-mode)
    (markdown-view-mode)))

#+end_src
